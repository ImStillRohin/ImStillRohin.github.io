Number Guessing Game

import random from pyscript import Element random_number = random.randint(1, 100) def checkGuess(event=None): input_box = Element("guessInput") guess = int(input_box.element.value) result_message = Element("resultMessage") if guess < random_number: result_message.write("Too low! Try again.") elif guess > random_number: result_message.write("Too high! Try again.") else: result_message.write("Congratulations! You guessed it right.") Element("guessButton").element.onclick = checkGuess
